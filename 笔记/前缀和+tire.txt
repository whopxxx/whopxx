#include<bits/stdc++.h>
using namespace std;
const int N = 3100000;
int son[N][2],cnt[N],idx;
int a[N],s[N];
void insert(int x,int v){
    int p = 0;
    for(int i = 30; ~i; i--){//0~31å…±32ä½
        int u = x >> i & 1;
        if(!son[p][u])son[p][u] = ++idx;
        p = son[p][u];
        cnt[p] += v;
    }
}

int ask(int x){
    int p = 0,res = 0;
    for(int i = 30; ~i; i--){
        int u = x >> i & 1;
        if(cnt[son[p][!u]])p = son[p][!u],res += 1 << i;
        else p = son[p][u]; 
    }
    return res;
}
int main(){
    int n,m;
    cin >> n >> m;
    for(int i = 1; i <= n; i++){
        int x;cin >> x;
        s[i] = s[i - 1]^x;
    }
    int ans = 0;
    insert(0,1);
    for(int i = 1; i <= n; i++){
        if(i > m)insert(s[i - m - 1],-1);
        ans = max(ans,ask(s[i]));
        insert(s[i],1);
    }
    cout << ans << endl;
    return 0;
}
